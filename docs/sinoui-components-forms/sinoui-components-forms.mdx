# @sinoui/sinoui-components-forms

以`sinoui-components`UI 库和[@sinoui/web-forms](https://github.com/sinoui/sinoui-forms/tree/master/packages/web-forms)为 UI 基础,
以[@sinoui/rx-form-state](https://github.com/sinoui/sinoui-forms/tree/master/packages/rx-form-state)库做状态管理所做的表单控件集合。
主要有以下几个常用的表单控件：

- [TextInput](/text-input) 输入框
- [Select](/select) 选择框
- [RadioGroup](/radio-group) 单选按钮组
- [Checkbox](/checkbox) 复选框
- [CheckboxGroup](/checkbox-group) 复选按钮组
- [DatePicker](/datepicker) 日期选择组件

以上组件的属性及具体使用方式详见各组件 API 文档。

## 依赖安装

```shell
yarn add @sinoui/sinoui-components-forms
```

或

```shell
npm install @sinoui/sinoui-components-forms
```

## 引入方式

```tsx
import {
  Select,
  SelectFieldProps,
  TextInput,
  TextInputFieldProps,
  RadioGroup,
  RadioGroupFieldProps,
  Radio,
  Checkbox,
  CheckboxFieldProps,
  CheckboxGroup,
  CheckboxGroupFieldProps,
  DatePicker,
  DatePickerFieldProps,
} from '@sinoui/sinoui-components-forms';
```

## 使用方式

基本用法：

```tsx
import React from 'react';
import {
  useFormState,
  FormValueMonitor,
  useFormSubmitting,
} from '@sinoui/rx-form-state';
import {
  Form,
  FormItem,
  Label,
  Select,
  TextInput,
  RadioGroup,
  Radio,
  Checkbox,
  CheckboxGroup,
  DatePicker,
} from '@sinoui/sinoui-components-forms';
import { Row, Column } from 'sinoui-components/Grid';
import { LoadingButton } from 'sinoui-components/Button';

const onSubmit = (values) => {
  // 使用setTimeout模拟向后端提交表单。
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve();
      alert('提交表单成功');
    }, 1000);
  });
};
function FormDemo() {
  const formState = useFormState(undefined, { onSubmit });
  const isSubmitting = useFormSubmitting(formState);
  return (
    <Form formState={formState}>
      <Row>
        <Column flex={12}>
          <FormItem>
            <Label width="120px">姓名</Label>
            <TextInput
              name="userName"
              required
              maxLength={10}
              placeholder="请输入姓名，如张三"
            />
          </FormItem>
        </Column>
        <Column flex={12}>
          <FormItem>
            <Label width="120px">性别</Label>
            <RadioGroup
              name="sex"
              validate={(value, values) =>
                value !== '男' && value !== '女' && '必须是男或者女'
              }
            >
              <Radio value="男">男</Radio>
              <Radio value="女">女</Radio>
            </RadioGroup>
          </FormItem>
        </Column>
        <Column flex={12}>
          <FormItem>
            <Label width="120px">出生日期</Label>
            <DatePicker name="birthday" />
          </FormItem>
        </Column>
        <Column flex={12}>
          <FormItem>
            <Label width="120px">年龄</Label>
            <TextInput name="age" type="number" endComponent="岁" />
          </FormItem>
        </Column>
        <Column flex={12}>
          <FormItem name="fav">
            <Label>兴趣爱好</Label>
            <CheckboxGroup name="fav" required>
              <Checkbox value="篮球">篮球</Checkbox>
              <Checkbox value="乒乓球">乒乓球</Checkbox>
              <Checkbox value="足球">足球</Checkbox>
              <Checkbox value="排球">排球</Checkbox>
            </CheckboxGroup>
          </FormItem>
        </Column>
        <Column flex={12}>
          <FormItem name="address">
            <Label>所在城市</Label>
            <Select name="address">
              <option value="">请选择</option>
              <option value="北京">北京</option>
              <option value="上海">上海</option>
              <option value="广州">广州</option>
              <option value="深圳">深圳</option>
            </Select>
          </FormItem>
        </Column>
      </Row>
      <FormValueMonitor>
        {(values) => <div>表单值：{JSON.stringify(values)}</div>}
      </FormValueMonitor>
      <LoadingButton
        type="submit"
        raised
        color="primary"
        loading={isSubmitting}
      >
        提交表单
      </LoadingButton>
    </Form>
  );
}
```
